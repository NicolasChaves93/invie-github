CREATE TABLE TOTEMS.SERVICIOS (
    SERVICIO_ID   VARCHAR2(100) NOT NULL, PK
    DESCRIPCION   VARCHAR2(1000) NOT NULL,
  USUARIO_CREACION VARCHAR(30) NOT NULL,
  FECHA_CREACION  TIMESTAMP NOT NULL,
  USUARIO_MODIFICACION VARCHAR2(30),
  FECHA_MODIFICACION TIMESTAMP
);

CREATE TABLE TOTEMS.PROPIEDADES_SERVICIOS (
    SERVICIO_ID          VARCHAR2(100) NOT NULL, PK
    PROPIEDAD_ID   		 VARCHAR2(500) NOT NULL, PK
    PROPIEDAD_VALOR      VARCHAR2(1000) NOT NULL,
	USUARIO_CREACION     VARCHAR(30) NOT NULL,
	FECHA_CREACION  	 TIMESTAMP NOT NULL,
	USUARIO_MODIFICACION VARCHAR2(30),
	FECHA_MODIFICACION   TIMESTAMP
);
